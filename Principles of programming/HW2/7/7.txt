//******************  Greatest Common Divisor *************************
int GCD( int x , int y )
{
	if( x == y )	return x ;
	else
	{
		if( x > y )				GCD( y , x-y ) ;
		else if( x < y )		GCD( x , y-x ) ;
	}
}
//***************** Least Common Multiple ******************************
int LCM( int x , int y )
{
	return (x*y) / GCD( x , y ) ;
}
