#include <iostream>
#include <conio.h>
#include <math.h>
using namespace std;
int main()
{
	long long n , sum = 0 , temp ;
	double i = 0 ;
	cout << "Pls enter binary number :" ;
	cin >> n ;
	temp = n ;

	while( temp > 0 )
	{
		sum += (temp%10) * pow( 2 , i) ;
		i++ ;
		temp /= 10 ;
	}
	
	cout << "(" << n << ")2" << " = " << "(" << sum << ")10" << endl ;
	
	getch();
	return 0 ;
}
